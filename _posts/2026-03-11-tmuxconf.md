~/.tmux.conf

```
# Send prefix
#set-option -g prefix C-a
#unbind-key C-a
#bind-key C-a send-prefix

# Use Ctrl-arrow keys to switch panes
bind -n C-Left select-pane -L
bind -n C-Right select-pane -R
bind -n C-Up select-pane -U
bind -n C-Down select-pane -D

# use Ctrl-jk to switch panes in loop
bind -n C-j select-pane -t :.+
bind -n C-k select-pane -t :.-

# Use Alt-arrow to reszie the pane by 10
bind -n M-Up    resize-pane -U 10
bind -n M-Down  resize-pane -D 10
bind -n M-Left  resize-pane -L 10
bind -n M-Right resize-pane -R 10

# Ctrl-hl to switch windows
bind -n C-h previous-window
bind -n C-l next-window

# Set easier window split keys
bind-key v split-window -h
bind-key h split-window -v

# Mouse mode
set -g mouse on

# disable mouse copy-mode jump to bottom automaticlly
bind-key    -T copy-mode    C-w                   send-keys -X copy-selection
bind-key    -T copy-mode    MouseDragEnd1Pane     send-keys -X copy-selection
bind-key    -T copy-mode    M-w                   send-keys -X copy-selection
bind-key    -T copy-mode-vi C-j                   send-keys -X copy-selection
bind-key    -T copy-mode-vi Enter                 send-keys -X copy-selection
bind-key    -T copy-mode-vi MouseDragEnd1Pane     send-keys -X copy-selection

# set mode-keys to vim
set -g mode-keys vi

# Easy config reload
bind-key r source-file ~/.tmux.conf \; display-message "tmux.conf reloaded"

# esc time in vim
set -g escape-time 10

set-option -g history-limit 50000

# send xterm format keys singnal to vim, like home/end
set -g xterm-keys on

# status bar content
# status left style "[session name] HH:MM |"
set -g status-left "#[fg=white][#S] #[fg=white]%H:%M #[fg=white]| "
set -g status-left-length 50

# status right style "[host name] CPU:x.x% RAM:x.x% yyyy-mm-dd"
set -g status-right "#[fg=white][#H] #[fg=green]CPU:#(top -bn1 | grep 'Cpu(s)' | awk '{printf \"%.1f%%\", $2}') #[fg=yellow]RAM:#(free -m | awk '/Mem/ {printf \"%.1f%%\", $3/$2*100}') #[fg=white]%Y-%m-%d"
set -g status-right-length 50

# color configuration
# status bg & fg(font color)
set -g status-style "bg=colour59,fg=colour255"

# status left color
# set -g status-left-style "bg=colour240,fg=colour255"

# status right color
# set -g status-right-style "bg=colour240,fg=colour255"

# windows list color
# set -g window-status-style "bg=colour236,fg=colour250"
set -g window-status-current-style "bg=colour202,fg=colour255"

# panel border color, default: gary
set -g pane-border-style "fg=colour240,bg=default"

# actived panel border color, default: transparent
set -g pane-active-border-style "fg=colour202,bg=default"

# windows list position
# set -g status-justify centre # left/right

```
