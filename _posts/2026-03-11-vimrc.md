~/.vimrc

```
" 开启行号
set number

" 启用状态栏
set laststatus=2

" 显示当前编辑状态
set showmode

" Mode Settings
let &t_SI.="\e[6 q\<Esc>]12;orange\x7" "SI = INSERT mode
let &t_SR.="\e[4 q\<Esc>]12;orange\x7" "SR = REPLACE mode
let &t_EI.="\e[2 q\<Esc>]12;orange\x7" "EI = NORMAL mode (ELSE)

" Cursor settings:

"  1 -> blinking block
"  2 -> solid block
"  3 -> blinking underscore
"  4 -> solid underscore
"  5 -> blinking vertical bar
"  6 -> solid vertical bar

" 开启鼠标模式
set mouse=a

" tab修改为4空格
set ts=4
set expandtab

" 设置大小写不敏感搜索
set ignorecase 
set smartcase

" 逐步显示搜索结果
set incsearch 
" 高亮搜索结果
set hlsearch
```
